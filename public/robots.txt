import { MetadataRoute } from 'next';
import { createClient } from '@supabase/supabase-js';

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!
);

export default async function sitemap(): Promise<MetadataRoute.Sitemap> {
  const baseUrl = 'https://riterapublishing.com';
  
  // Static pages
  const staticPages = [
    '',
    '/aboutus',
    '/people-behind-ritera',
    '/packages',
    '/books',
    '/blog',
    '/litspace',
    '/litspace/submit',
    '/careers',
    '/employee',
    '/contact',
    '/privacy-policy',
    '/terms-of-service',
  ].map((route) => ({
    url: `${baseUrl}${route}`,
    lastModified: new Date(),
    changeFrequency: 'weekly' as const,
    priority: route === '' ? 1 : 0.8,
  }));
  
  // Fetch dynamic blog posts
  const { data: posts } = await supabase
    .from('blog_posts')
    .select('slug, updated_at')
    .eq('published', true);
  
  const blogPages = (posts || []).map((post) => ({
    url: `${baseUrl}/blog/${post.slug}`,
    lastModified: new Date(post.updated_at),
    changeFrequency: 'monthly' as const,
    priority: 0.7,
  }));
  
  // Fetch books
  const { data: books } = await supabase
    .from('books')
    .select('slug, updated_at')
    .eq('featured', true);
  
  const bookPages = (books || []).map((book) => ({
    url: `${baseUrl}/books/${book.slug}`,
    lastModified: new Date(book.updated_at),
    changeFrequency: 'monthly' as const,
    priority: 0.7,
  }));
  
  // Fetch litspace posts
  const { data: litspace } = await supabase
    .from('litspace_posts')
    .select('slug, updated_at')
    .eq('approved', true);
  
  const litspacePages = (litspace || []).map((post) => ({
    url: `${baseUrl}/litspace/${post.slug}`,
    lastModified: new Date(post.updated_at),
    changeFrequency: 'monthly' as const,
    priority: 0.6,
  }));
  
  // Fetch careers
  const { data: careers } = await supabase
    .from('careers')
    .select('slug, updated_at')
    .eq('status', 'Active');
  
  const careerPages = (careers || []).map((job) => ({
    url: `${baseUrl}/careers/${job.slug}`,
    lastModified: new Date(job.updated_at),
    changeFrequency: 'weekly' as const,
    priority: 0.7,
  }));
  
  // Fetch authors
  const { data: authors } = await supabase
    .from('authors')
    .select('slug, updated_at');
  
  const authorPages = (authors || []).map((author) => ({
    url: `${baseUrl}/authors/${author.slug}`,
    lastModified: new Date(author.updated_at),
    changeFrequency: 'monthly' as const,
    priority: 0.6,
  }));
  
  return [
    ...staticPages,
    ...blogPages,
    ...bookPages,
    ...litspacePages,
    ...careerPages,
    ...authorPages,
  ];
}
